    (function () {
      const canvas = document.getElementById("fireworksCanvas");

      const fireworks = [];

      let fireworksEnabled = false;

      class Firework {
        constructor(x, y, targetY, color) {
          this.x = x;
          this.y = y;
          this.targetY = targetY;
          this.speed = (Math.random() * 4) + 6;
          this.angle = -Math.PI / 2;
          this.vx = Math.cos(this.angle) * this.speed;
          this.vy = Math.sin(this.angle) * this.speed;
          this.color = color;
          this.trail = [];
          this.done = false;
        }

        update() {
          this.trail.push({ x: this.x, y: this.y });
          if (this.trail.length > 10) this.trail.shift();

          this.x += this.vx;
          this.y += this.vy;
          this.vy += 0.07;

          if (this.vy >= 0 || this.y <= this.targetY) {
            this.explode();
            this.done = true;
          }
        }

        explode() {
          const count = 48 + Math.floor(Math.random() * 28);
          for (let i = 0; i < count; i++) {
            const angle = (Math.PI * 2 * i) / count;
            const speed = Math.random() * 3 + 2;
            particles.push(new Particle(this.x, this.y, angle, speed, this.color));
          }
        }

        draw() {
          ctx.beginPath();
          const last = this.trail[0];
          ctx.moveTo(this.x, this.y);
          if (last) ctx.lineTo(last.x, last.y);
          ctx.strokeStyle = this.color;
          ctx.lineWidth = 2;
          ctx.stroke();
        }
      }

      class Particle {
        constructor(x, y, angle, speed, color) {
          this.x = x;
          this.y = y;
          this.vx = Math.cos(angle) * speed;
          this.vy = Math.sin(angle) * speed;
          this.friction = 0.97;
          this.gravity = 0.08;
          this.color = color;
          this.opacity = 1;
          this.decay = Math.random() * 0.015 + 0.007;
          this.size = Math.random() * 1.8 + 0.7;
        }

        update() {
          this.vx *= this.friction;
          this.vy *= this.friction;
          this.vy += this.gravity;
          this.x += this.vx;
          this.y += this.vy;
          this.opacity -= this.decay;
        }

        draw() {
          ctx.save();
          ctx.globalAlpha = this.opacity;
          ctx.beginPath();
          ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
          ctx.fillStyle = this.color;
          ctx.fill();
          ctx.restore();
        }
      }

        ctx.fillRect(0, 0, width, height);
        ctx.globalCompositeOperation = "lighter";

        if (!lastLaunch || timestamp - lastLaunch >= launchInterval) {
          const startX = Math.random() * width * 0.8 + width * 0.1;
          const targetY = Math.random() * height * 0.4 + height * 0.1;
          fireworks.push(new Firework(startX, height + 10, targetY, randomColor()));
          lastLaunch = timestamp;
        }

        for (let i = fireworks.length - 1; i >= 0; i--) {
          fireworks[i].update();
          fireworks[i].draw();
          if (fireworks[i].done) fireworks.splice(i, 1);
        }

        for (let i = particles.length - 1; i >= 0; i--) {
          particles[i].update();
          particles[i].draw();
          if (particles[i].opacity <= 0) particles.splice(i, 1);
        }
      }

      requestAnimationFrame(animate);

      window.startFireworks = function () {
        fireworksEnabled = true;
        lastLaunch = 0;
      };
    })();
  </script>
